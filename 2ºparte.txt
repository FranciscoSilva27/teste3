FUNÇÃO verificar_numero(numero_secreto, palpite)
    INÍCIO

        // Calcula a diferença entre o número secreto e o palpite
        diferenca ? ABS(numero_secreto - palpite)

        // Fornece pistas com base na diferença
        SE diferenca > 25 ENTÃO
            ESCREVER "Frio"
        SENÃO SE diferenca >= 5 E diferenca <= 25 ENTÃO
            ESCREVER "Morno"
        SENÃO SE diferenca < 5 ENTÃO
            ESCREVER "Quente"
        FIM SE

        // Verifica se o jogador acertou
        SE palpite = numero_secreto ENTÃO
            ESCREVER "Parabéns! Você acertou o número secreto!"
        FIM SE

    FIM
FIM FUNÇÃO
